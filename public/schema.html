  %% ── Multi-tenancy ─────────────────────────────────────────
  tenants {
    uuid id PK
    string name
    string domain
    string status
    string plan
  }
  tenant_members {
    uuid id PK
    uuid tenant_id FK
    uuid user_id FK
    string role
  }
  tenants ||--o{ tenant_members : "has"

  %% ── CRM Core ──────────────────────────────────────────────
  accounts {
    uuid id PK
    uuid tenant_id FK
    uuid parent_account_id FK
    string name
    string account_number
    string industry
    string account_type
    string status
    string phone
    string email
    string address_city
    string address_state
  }
  tenants    ||--o{ accounts : "owns"
  accounts   ||--o{ accounts : "parent >"

  contacts {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid original_lead_id FK
    string first_name
    string last_name
    string email
    string phone
    string job_title
  }
  accounts ||--o{ contacts : "has"

  leads {
    uuid id PK
    uuid tenant_id FK
    uuid converted_account_id FK
    uuid converted_contact_id FK
    string first_name
    string last_name
    string email
    string company
    string status
    string lead_source
  }
  leads |o--o| accounts  : "converts to"
  leads |o--o| contacts  : "converts to"

  connections {
    uuid id PK
    uuid contact_id FK
    uuid connected_contact_id FK
    string relationship_type
  }
  contacts ||--o{ connections : "from"
  contacts ||--o{ connections : "to"

  %% ── Sales Pipeline ────────────────────────────────────────
  opportunities {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid contact_id FK
    string name
    string stage
    decimal amount
    int probability
    date close_date
    string status
  }
  accounts     ||--o{ opportunities : "has"
  contacts     ||--o{ opportunities : "primary contact"

  quotes {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid opportunity_id FK
    string name
    string quote_number
    string status
    decimal total
  }
  opportunities ||--o{ quotes : "generates"
  accounts      ||--o{ quotes : "billed to"

  products {
    uuid id PK
    uuid tenant_id FK
    string name
    string sku
    decimal unit_price
    string type
  }

  quote_lines {
    uuid id PK
    uuid quote_id FK
    uuid product_id FK
    int quantity
    decimal unit_price
    decimal total_price
  }
  quotes    ||--o{ quote_lines : "contains"
  products  ||--o{ quote_lines : "priced via"

  commission_splits {
    uuid id PK
    uuid tenant_id FK
    uuid opportunity_id FK
    uuid contract_id FK
    uuid contact_id FK
    decimal split_percentage
    decimal amount
    string role
    string status
  }
  opportunities ||--o{ commission_splits : "splits"
  contacts      ||--o{ commission_splits : "assigned to"

  %% ── Contracts & Energy ────────────────────────────────────
  contracts {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid opportunity_id FK
    uuid parent_contract_id FK
    string name
    string contract_number
    string contract_type
    string contract_status
    string service_status
    decimal value
    date contract_start_date
    date billing_schedule_end_date
  }
  accounts      ||--o{ contracts     : "has"
  opportunities ||--o{ contracts     : "closes as"
  contracts     ||--o{ contracts     : "parent >"

  energy_programs {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid opportunity_id FK
    string name
    string pgm_id
    string salesforce_id
    string status
    date contract_start_date
    date billing_schedule_end_date
  }
  accounts      ||--o{ energy_programs : "enrolled in"
  opportunities ||--o{ energy_programs : "leads to"
  contracts     ||--o{ energy_programs : "governs"

  measures {
    uuid id PK
    uuid tenant_id FK
    uuid energy_program_id FK
    uuid account_id FK
    string name
    string measure_type
    string status
    decimal estimated_savings
    decimal actual_savings
  }
  energy_programs ||--o{ measures : "includes"

  %% ── Projects ──────────────────────────────────────────────
  projects {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid contract_id FK
    uuid energy_program_id FK
    string name
    string status
    string priority
    decimal budget
    int progress_percent
    date start_date
    date end_date
  }
  accounts        ||--o{ projects : "owns"
  contracts       ||--o{ projects : "under"
  energy_programs ||--o{ projects : "executes"

  project_milestones {
    uuid id PK
    uuid project_id FK
    string name
    date due_date
    string status
    int sort_order
  }
  projects ||--o{ project_milestones : "has"

  project_tasks {
    uuid id PK
    uuid project_id FK
    uuid milestone_id FK
    uuid parent_task_id FK
    string name
    string status
    string priority
    decimal estimated_hours
    decimal actual_hours
  }
  projects           ||--o{ project_tasks       : "has"
  project_milestones ||--o{ project_tasks       : "groups"
  project_tasks      ||--o{ project_tasks       : "subtask >"

  buildings {
    uuid id PK
    uuid tenant_id FK
    uuid account_id FK
    uuid energy_program_id FK
    uuid project_id FK
    string name
    string building_type
    string address_city
    string address_state
    int square_footage
    int energy_star_score
    string status
  }
  accounts        ||--o{ buildings : "owns"
  energy_programs ||--o{ buildings : "covers"
  projects        ||--o{ buildings : "scoped to"

  %% ── Finance & Invoicing ───────────────────────────────────
  invoices {

    string name
    string status

  invoice_items {

  }
  invoices        ||--o{ invoice_items : "has"

  energy_programs ||--o{ budget_items : "tracks"
